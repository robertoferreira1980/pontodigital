async function doPunch(force=false) { if (isSending || pendingRetry) return; if (!navigator.onLine) { toast('❌ Sem internet!','error'); return; } if (!currentEmpId) return; const next = nextSlot(currentEmpId); if (!next) return; const emp = employees.find(e => e.id === currentEmpId); const loc = locations.find(l => l.id === emp.locId); if (!loc) { toast('⚠️ Local não definido para este funcionário','warn'); return; } isSending = true; const btn = document.getElementById('main-btn'); btn.disabled = true; document.getElementById('btn-override').style.display = 'none'; setStatus('checking','📡 Verificando GPS...'); try { const pos = await getPos(); const dist = calcDist(pos.coords.latitude, pos.coords.longitude, loc.lat, loc.lng); if (dist > loc.radius && !force) { setStatus('far', `🚫 Você está a ${Math.round(dist)}m do local. Raio: ${loc.radius}m.`); if (settings.allowOverride !== false) document.getElementById('btn-override').style.display = 'block'; isSending = false; btn.disabled = false; return; } const now = new Date(); const rec = { id: Date.now(), empId: currentEmpId, empName: emp.name, slot: next, type: next.includes('entrada') ? 'entrada' : 'saida', datetime: now.toISOString(), date: now.toLocaleDateString('pt-BR'), time: now.toLocaleTimeString('pt-BR',{hour:'2-digit',minute:'2-digit',second:'2-digit'}), lat: pos.coords.latitude.toFixed(6), lng: pos.coords.longitude.toFixed(6), dist: force ? `⚠️${Math.round(dist)}` : Math.round(dist), synced: false }; // 3. O aplicativo NÃO exibe e NÃO salva localmente ainda, apenas envia. setStatus('checking', '📡 Enviando para a planilha...'); const cloudResult = await callCloud({ action: 'addRecord', record: rec }, 3); if (cloudResult.local) { // Sem URL configurada — modo offline local rec.synced = false; records.unshift(rec); saveAll(); setStatus('ok', `✅ Registrado localmente às ${rec.time.slice(0,5)}`); toast('✅ Ponto registrado offline!','success'); updatePunchBtn(); updateSlots(); } else if (cloudResult.ok) { // Recebemos resposta da planilha! const responseData = cloudResult.data.data; // Pega o retorno lá do Code.gs if (responseData && responseData.status === 'duplicate') { // Alguém já bateu esse ponto em outro celular rec.time = responseData.time; // Pega o horário que já estava gravado lá rec.synced = true; // Remove duplicatas locais (se houver) e salva o horário oficial records = records.filter(r => !(r.empId === rec.empId && r.date === rec.date && r.slot === rec.slot)); records.unshift(rec); saveAll(); setStatus('warn', `⚠️ Já registrado por outro aparelho às ${rec.time.slice(0,5)}`); toast('Ponto recuperado da planilha!', 'warn'); updatePunchBtn(); updateSlots(); } else { // Gravado com sucesso na planilha if (responseData && responseData.time) { rec.time = responseData.time; // Garante o horário exato retornado } rec.synced = true; if (!records.find(r => String(r.id) === String(rec.id))) { records.unshift(rec); } saveAll(); setStatus('ok', `✅ Registrado às ${rec.time.slice(0,5)} ☁️`); toast('✅ Ponto salvo na planilha!','success'); updatePunchBtn(); updateSlots(); } // Sincroniza o resto em background syncCloud(false); } else { // Falha de rede: aí sim ele guarda na fila de espera local pendingRetry = rec; pending.push(rec); saveAll(); setStatus('warn', `⚠️ Sem resposta da rede. Horário ${rec.time.slice(0,5)} na fila — clique Reenviar.`); toast('❌ Falha ao enviar. Toque Reenviar quando tiver sinal.','error'); updatePunchBtn(); updateSlots(); } } catch(e) { handleGeoErr(e); } isSending = false; if (!pendingRetry) btn.disabled = false; }
